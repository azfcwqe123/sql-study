```SQL
SELECT
    COUNT(*) AS COUNT
FROM
    ECOLI_DATA
WHERE
    GENOTYPE & 2 = 0 # 2번 형질을 가지지 않음
    AND (GENOTYPE & 1 = 1 OR GENOTYPE & 4 = 4) # 1번, 3번 형질을 가짐
```
